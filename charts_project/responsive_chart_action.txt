        function getOption() {
            var w = window.innerWidth;

            return {
                title: { text: '遇到网暴，你会怎么做？', left: 'center', top: '5%', textStyle:{fontSize: isMobile?16:24} },
                grid: { 
                    left: '5%', right: isMobile?'15%':'10%', 
                    bottom: '5%', top: '15%', containLabel: true 
                },
                xAxis: { type: 'value', show: false },
                yAxis: { 
                    type: 'category', 
                    data: ['跟风评论', '围观不语', '理性劝解', '举报言论'],
                    axisLabel: { fontSize: isMobile?13:16, margin: 10 }
                },
                series: [{
                    type: 'bar',
                    barWidth: isMobile ? 20 : 40,
                    data: [4, 375, 375, 422],
                    itemStyle: {
                        borderRadius: [0, 20, 20, 0],
                        color: function(p) { return ['#ccc','#FAC858','#73C0DE','#EE6666'][p.dataIndex]; }
                    },
                    label: { show: true, position: 'right', fontSize: isMobile?12:14 }
                }]
            };
        }
        chart.setOption(getOption());
        window.onresize = function() { chart.resize(); chart.setOption(getOption()); };
    </script>
